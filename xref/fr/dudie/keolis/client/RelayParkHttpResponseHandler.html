19  /**
20   * Handles http responses containing relay parks in json format.
21   * 
22   * <pre>
23   * "relaypark": [
24   *     {
25   *         "name": "La Poterie"
26   *         "latitude": "48.0868139293"
27   *         "longitude": "-1.6434973209"
28   *         "carparkavailable": "249"
29   *         "carparkcapacity": "388"
30   *         "lastupdate": "2011-09-03T18:26:11+02:00"
31   *         "state": "0"
32   *         "distance": "5184.2893316804"
33   *     }
34   * ]
35   * </pre>
36   * 
37   * @author Jérémie Huchet
38   */
39  public class RelayParkHttpResponseHandler implements ResponseHandler<List<RelayPark>> {
40  
41      /** The event logger. */
42      private static final Logger LOGGER = LoggerFactory
43              .getLogger(RelayParkHttpResponseHandler.class);
44  
45      /** Default character encoding to use. */
46      private static final String DEFAULT_ENCODING = "utf-8";
47  
48      /**
49       * {@inheritDoc}
50       * 
51       * @see org.apache.http.client.ResponseHandler#handleResponse(org.apache.http.HttpResponse)
52       */
53      @Override
54      public List<RelayPark> handleResponse(final HttpResponse response)
55              throws ClientProtocolException, IOException {
56  
57          if (LOGGER.isDebugEnabled()) {
58              LOGGER.debug("handleResponse.start");
59          }
60          final String content = EntityUtils.toString(response.getEntity(), DEFAULT_ENCODING);
61  
62          JSONObject data = null;
63          try {
64              data = KeoUtils.getServiceResponse(content);
65          } catch (final JSONException e) {
66              throw new IOException("Unable to parse the json response received from Keolis:\n"
67                      + content);
68          }
69  
70          List<RelayPark> listRelayParks = null;
71  
72          if (null != data) {
73              // else handle multiple stations
74              final JSONArray jsonParks = data.optJSONArray("relaypark");
75              if (null != jsonParks) {
76                  listRelayParks = new ArrayList<RelayPark>();
77                  for (int i = 0; !jsonParks.isNull(i); i++) {
78                      listRelayParks.add(convertJsonObjectToRelayPark(jsonParks.optJSONObject(i)));
79                  }
80              } else {
81                  if (LOGGER.isDebugEnabled()) {
82                      LOGGER.debug("0 relay park found in response");
83                  }
84              }
85          }
86  
87          if (LOGGER.isDebugEnabled()) {
88              LOGGER.debug("handleResponse.end");
89          }
90          return listRelayParks;
91      }
92  
93      /**
94       * Converts a json formatted relay park to an Object representation.
95       * 
96       * @param jsonPark
97       *            the json representation
98       * @return an object representation
99       */
100     private RelayPark convertJsonObjectToRelayPark(final JSONObject jsonPark) {
101 
102         final RelayPark park = new RelayPark();
103         park.setName(jsonPark.optString("name"));
104         park.setCarParkAvailable(jsonPark.optInt("carparkavailable"));
105         park.setCarParkCapacity(jsonPark.optInt("carparkcapacity"));
106         park.setLastUpdate(KeoUtils.convertJsonStringToDate(jsonPark.optString("lastupdate")));
107         park.setLatitude((int) (jsonPark.optDouble("latitude") * 1E6));
108         park.setLongitude((int) (jsonPark.optDouble("longitude") * 1E6));
109         park.setState(jsonPark.optInt("state"));
110         return park;
111     }
