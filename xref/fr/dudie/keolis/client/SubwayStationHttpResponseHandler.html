21   * Handles http responses containing subway stations in json format.
22   * 
23   * <pre>
24   * "station":[
25   *    {
26   *       "id":"ANF",
27   *       "name":"Anatole France",
28   *       "latitude":"48.11810000",
29   *       "longitude":"-1.687460000",
30   *       "hasPlatformDirection1":"1",
31   *       "hasPlatformDirection2":"1",
32   *       "rankingPlatformDirection1":"12",
33   *       "rankingPlatformDirection2":"18",
34   *       "floors":"-1",
35   *       "lastupdate":"2010-11-23T23:15:58+01:00"
36   *    }
37   * ]
38   * </pre>
39   * 
40   * @author Jérémie Huchet
41   */
42  public final class SubwayStationHttpResponseHandler implements ResponseHandler<List<SubwayStation>> {
43  
44      /** The event logger. */
45      private static final Logger LOGGER = LoggerFactory
46              .getLogger(SubwayStationHttpResponseHandler.class);
47  
48      /** Default character encoding to use. */
49      private static final String DEFAULT_ENCODING = "utf-8";
50  
51      /**
52       * {@inheritDoc}
53       * 
54       * @see org.apache.http.client.ResponseHandler#handleResponse(org.apache.http.HttpResponse)
55       */
56      @Override
57      public List<SubwayStation> handleResponse(final HttpResponse response)
58              throws ClientProtocolException, IOException {
59  
60          if (LOGGER.isDebugEnabled()) {
61              LOGGER.debug("handleResponse.start");
62          }
63          final String content = EntityUtils.toString(response.getEntity(), DEFAULT_ENCODING);
64  
65          JSONObject data = null;
66          try {
67              data = KeoUtils.getServiceResponse(content);
68          } catch (final JSONException e) {
69              throw new IOException("Unable to parse the json response received from Keolis:\n"
70                      + content);
71          }
72  
73          List<SubwayStation> listStations = null;
74  
75          if (null != data) {
76              // try to handle the response as if there is one station
77              final JSONObject jsonStation = data.optJSONObject("station");
78              if (null != jsonStation) {
79                  if (LOGGER.isDebugEnabled()) {
80                      LOGGER.debug("response contains 1 station");
81                  }
82                  listStations = new ArrayList<SubwayStation>(1);
83                  listStations.add(convertJsonObjectToSubwayStation(jsonStation));
84              } else {
85                  // else handle multiple stations
86                  final JSONArray jsonStations = data.optJSONArray("station");
87                  if (null != jsonStations) {
88                      if (LOGGER.isDebugEnabled()) {
89                          LOGGER.debug("response contains multiple stations");
90                      }
91                      listStations = new ArrayList<SubwayStation>();
92                      for (int i = 0; !jsonStations.isNull(i); i++) {
93                          listStations.add(convertJsonObjectToSubwayStation(jsonStations
94                                  .optJSONObject(i)));
95                      }
96                  }
97              }
98          }
99          if (LOGGER.isDebugEnabled()) {
100             LOGGER.debug("handleResponse.end");
101         }
102         return listStations;
103     }

105     /**
106      * Converts a json object to a bean representing a subway station.
107      * 
108      * @param jsonObject
109      *            the json object to convert to a subway station
110      * @return the subway station bean
111      */
112     private fr.dudie.keolis.model.SubwayStation convertJsonObjectToSubwayStation(
113             final JSONObject jsonObject) {
114 
115         final SubwayStation station = new SubwayStation();
116         station.setId(jsonObject.optString("id"));
117         station.setName(StringUtils.capitalize(jsonObject.optString("name")));
118         station.setLongitude((int) (jsonObject.optDouble("longitude") * 1E6));
119         station.setLatitude((int) (jsonObject.optDouble("latitude") * 1E6));
120         station.setHasPlatformDirection1(KeoUtils.convertJsonIntToBoolean(jsonObject
121                 .optInt("hasPlatformDirection1")));
122         station.setHasPlatformDirection2(KeoUtils.convertJsonIntToBoolean(jsonObject
123                 .optInt("hasPlatformDirection2")));
124         String rankPlatformDir = jsonObject.optString("rankingPlatformDirection1");
125         if (null != rankPlatformDir && !"".equals(rankPlatformDir)) {
126             station.setRankingPlatformDirection1(Integer.valueOf(rankPlatformDir));
127         } else {
128             station.setRankingPlatformDirection1(0);
129         }
130         rankPlatformDir = jsonObject.optString("rankingPlatformDirection2");
131         if (null != rankPlatformDir && !"".equals(rankPlatformDir)) {
132             station.setRankingPlatformDirection2(Integer.valueOf(rankPlatformDir));
133         } else {
134             station.setRankingPlatformDirection2(0);
135         }
136         station.setFloors(jsonObject.optInt("floors"));
137         station.setLastUpdate(KeoUtils.convertJsonStringToDate(jsonObject.optString("lastupdate")));
138 
139         return station;
140     }
